52  inline void GenerateUintKeys(const std::vector<std::reference_wrapper<const compute::vector<unsigned int>>> &keysFirst,
53  const compute::vector<unsigned int>::iterator &newKeysFirst,
54  std::size_t preferredWorkGroupSize,
55  compute::command_queue &queue) {
56  compute::detail::meta_kernel k("spla_reduce_by_key_new_key_flags");
57  const std::size_t count = keysFirst.at(0).get().size();
58  const std::size_t nKeys = keysFirst.size();
59  k.add_set_arg<const uint_>("count", static_cast<uint_>(count));
60 
61  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
62  << k.decl<uint_>("value") << " = 0;\n"
63  << "if(gid >= count){\n return;\n}\n"
64  << "if(gid > 0){ \n"
65  << DeclareMultiKey{"key", nKeys} << ";\n"
66  << AssignKey{KeyVar("key"), KeyVec(keysFirst, "gid", k), nKeys} << ";\n"
67  << DeclareMultiKey{"previous_key", nKeys} << ";\n"
68  << AssignKey{KeyVar("previous_key"), KeyVec(keysFirst, "gid - 1", k), nKeys}
69  << " value = " << CompareKey{KeyVar("previous_key"), KeyVar("key"), nKeys} << " ? 0 : 1;\n"
70  << "}\n else {\n"
71  << " value = 0;\n"
72  << "}\n"
73  << newKeysFirst[k.var<const uint_>("gid")] << " = value;\n";
74 
75  const compute::context &context = queue.get_context();
76  compute::kernel kernel = k.compile(context);
77 
78  auto workGroupsNo = static_cast<std::size_t>(
79  std::ceil(static_cast<float>(count) / static_cast<float>(preferredWorkGroupSize)));
80 
81  queue.enqueue_1d_range_kernel(kernel,
82  0,
83  workGroupsNo * preferredWorkGroupSize,
84  preferredWorkGroupSize);
85 
86  inclusive_scan(newKeysFirst, newKeysFirst + static_cast<std::ptrdiff_t>(count),
87  newKeysFirst, queue);
88  }

90  inline void CarryOuts(const compute::vector<uint_> &keys,
91  const compute::vector<unsigned char> &values,
92  const compute::vector<uint_>::iterator &carryOutsKeysFirst,
93  const compute::vector<unsigned char>::iterator &carryOutValuesFirst,
94  std::size_t workGroupSize,
95  std::size_t vBytes,
96  const std::string &reduceBody,
97  compute::command_queue &queue) {
98  compute::vector<uint_>::iterator keysFirst = keys.begin();
99  compute::vector<unsigned char>::iterator valuesFirst = values.begin();
100  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_outs");
101  k.add_set_arg<const uint_>("count", static_cast<uint_>(keys.size()));
102  size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
103  size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
104 
105  ReduceOp reduceOp(k, "spla_reduce", reduceBody, vBytes);
106 
107  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
108  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
109  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
110  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
111  << k.decl<uint_>("key") << ";\n"
112  << DeclareVal{"value", vBytes} << ";\n"
113  << "if(gid < count){\n"
114  << k.var<uint_>("key") << " = " << keysFirst[k.var<const uint_>("gid")] << ";\n"
115  << AssignVal{ValVar("value"), ValArrItem(valuesFirst, "gid", vBytes, k), vBytes} << ";\n"
116  << "lkeys[lid] = key;\n"
117  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("value"), vBytes} << ";\n"
118  << "}\n"
119  << DeclareVal{"result", vBytes} << ";\n"
120  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
121  << k.decl<uint_>("other_key") << ";\n"
122  << DeclareVal{"other_value", vBytes} << ";\n"
123  << "for (" << k.decl<uint_>("offset") << " = 1; offset < wg_size; offset *= 2) {\n"
124  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
125  << " if (lid >= offset){\n"
126  << " other_key = lkeys[lid - offset];\n"
127  << " if(other_key == key){\n"
128  << AssignVal{ValVar("other_value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
129  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
130  << " }\n"
131  << " }\n"
132  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
133  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
134  << "}\n"
135  << "if (lid == (wg_size - 1)){\n"
136  << carryOutsKeysFirst[k.var<const uint_>("group_id")] << " = key;\n"
137  << AssignVal{ValArrItem(carryOutValuesFirst, "group_id", vBytes, k), ValVar("result"), vBytes} << ";\n"
138  << "}\n";
139 
140 
141  auto workGroupsNo = static_cast<std::size_t>(
142  std::ceil(static_cast<float>(keys.size()) / static_cast<float>(workGroupSize)));
143 
144  const compute::context &context = queue.get_context();
145  compute::kernel kernel = k.compile(context);
146  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
147  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
148 
149  queue.enqueue_1d_range_kernel(kernel,
150  0,
151  workGroupsNo * workGroupSize,
152  workGroupSize);
153  }

155  inline void CarryIns(const compute::vector<uint_>::iterator &carryOutsKeysFirst,
156  const compute::vector<unsigned char>::iterator &carryOutValuesFirst,
157  const compute::vector<unsigned char>::iterator &carryInValuesFirst,
158  std::size_t carryOutSize,
159  std::size_t workGroupSize,
160  std::size_t vBytes,
161  const std::string &reduceBody,
162  compute::command_queue &queue) {
163  auto valuesPreWorkItem = static_cast<uint_>(
164  std::ceil(static_cast<float>(carryOutSize) / static_cast<float>(workGroupSize)));
165 
166  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_ins");
167  k.add_set_arg<const uint_>("carry_out_size", uint_(carryOutSize));
168  k.add_set_arg<const uint_>("values_per_work_item", valuesPreWorkItem);
169  std::size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
170  std::size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
171 
172  ReduceOp reduceOp(k, "spla_reduce", reduceBody, vBytes);
173 
174  k << k.decl<uint_>("id") << " = get_global_id(0) * values_per_work_item;\n"
175  << k.decl<uint_>("idx") << " = id;\n"
176  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
177  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
178  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
179  << k.decl<uint_>("key") << ";\n"
180  << DeclareVal{"value", vBytes} << ";\n"
181  << k.decl<uint_>("previous_key") << ";\n"
182  << DeclareVal{"result", vBytes} << ";\n"
183  << "if (id < carry_out_size) {\n"
184  << k.var<uint_>("previous_key") << " = " << carryOutsKeysFirst[k.var<const uint_>("id")] << ";\n"
185  << AssignVal{ValVar("result"), ValArrItem(carryOutValuesFirst, "id", vBytes, k), vBytes} << ";\n"
186  << AssignVal{ValArrItem(carryInValuesFirst, "id", vBytes, k), ValVar("result"), vBytes} << ";\n"
187  << "}\n"
188  << k.decl<const uint_>("end") << " = (id + values_per_work_item) <= carry_out_size"
189  << " ? (values_per_work_item + id) : carry_out_size;\n"
190  << "for (idx = idx + 1; idx < end; idx += 1) {\n"
191  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
192  << AssignVal{ValVar("value"), ValArrItem(carryOutValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
193  << " if(previous_key == key){\n"
194  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
195  << " }\n else { \n"
196  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
197  << " }\n"
198  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("result"), vBytes} << ";\n"
199  << " previous_key = key;\n"
200  "}\n"
201  << "lkeys[lid] = previous_key;\n"
202  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
203  << "for (" << k.decl<uint_>("offset") << " = 1; "
204  << "offset < wg_size; offset *= 2){\n"
205  " barrier(CLK_LOCAL_MEM_FENCE);\n"
206  << " if(lid >= offset){\n"
207  " key = lkeys[lid - offset];\n"
208  << " if(previous_key == key){\n"
209  << AssignVal{ValVar("value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
210  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
211  << " }\n"
212  << " }\n"
213  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
214  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
215  << "}\n"
216  << "barrier(CLK_LOCAL_MEM_FENCE);\n"
217  << "if(lid > 0){\n"
218  << " previous_key = lkeys[lid - 1];\n"
219  << AssignVal{ValVar("result"), ValArrItem("lvals", "lid - 1", vBytes), vBytes} << ";\n"
220  << "}\n"
221  << "for (idx = id; idx < id + values_per_work_item; idx += 1) {\n"
222  << " barrier( CLK_GLOBAL_MEM_FENCE );\n"
223  << " if (lid > 0 && idx < carry_out_size) {\n"
224  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
225  << AssignVal{ValVar("value"), ValArrItem(carryInValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
226  << " if(previous_key == key){\n"
227  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("value")) << ";\n"
228  << " }\n"
229  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("value"), vBytes} << ";\n"
230  << " }\n"
231  << "}\n";
232 
233  const compute::context &context = queue.get_context();
234  compute::kernel kernel = k.compile(context);
235  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
236  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
237 
238  queue.enqueue_1d_range_kernel(kernel,
239  0,
240  workGroupSize,
241  workGroupSize);

244  inline void FinalReduction(const std::vector<std::reference_wrapper<const compute::vector<uint_>>> &keys,
245  const compute::vector<unsigned char>::iterator &valuesFirst,
246  const std::vector<std::reference_wrapper<compute::vector<uint_>>> &keysResult,
247  const compute::vector<unsigned char>::iterator &valuesResult,
248  std::size_t count,
249  const compute::vector<uint_>::iterator &newKeysFirst,
250  const compute::vector<uint_>::iterator &carryInKeysFirst,
251  const compute::vector<unsigned char>::iterator &carryInValuesFirst,
252  std::size_t workGroupSize,
253  std::size_t vBytes,
254  const std::string &reduceBody,
255  compute::command_queue &queue) {
256  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_final_reduction");
257  k.add_set_arg<const uint_>("count", static_cast<uint_>(count));
258  const std::size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
259  const std::size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
260  const std::size_t nKeys = keys.size();
261  assert(nKeys == keysResult.size());
262  ReduceOp reduceOp(k, "spla_reduce", reduceBody, vBytes);
263 
264  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
265  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
266  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
267  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
268  << k.decl<uint_>("key") << ";\n"
269  << DeclareVal{"value", vBytes} << ";\n"
270  << "if (gid < count) {\n"
271  << k.var<uint_>("key") << " = " << newKeysFirst[k.var<const uint_>("gid")] << ";\n"
272  << AssignVal{ValVar("value"), ValArrItem(valuesFirst, "gid", vBytes, k), vBytes} << ";\n"
273  << "lkeys[lid] = key;\n"
274  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("value"), vBytes} << ";\n"
275  << "}\n"
276  << DeclareVal{"result", vBytes} << ";\n"
277  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
278  << k.decl<uint_>("other_key") << ";\n"
279  << DeclareVal{"other_value", vBytes} << ";\n"
280  << "for (" << k.decl<uint_>("offset") << " = 1; offset < wg_size ; offset *= 2) {\n"
281  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
282  << " if(lid >= offset) {\n"
283  << " other_key = lkeys[lid - offset];\n"
284  << " if(other_key == key){\n"
285  << AssignVal{ValVar("other_value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
286  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
287  << " }\n"
288  << " }\n"
289  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
290  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
291  << "}\n"
292  << "if (gid >= count) {\n return;\n};\n"
293  << k.decl<const bool>("save") << " = (gid < (count - 1)) ?"
294  << newKeysFirst[k.var<const uint_>("gid + 1")] << " != key : true;\n"
295  << k.decl<uint_>("carry_in_key") << ";\n"
296  << "if(group_id > 0 && save) {\n"
297  << " carry_in_key = " << carryInKeysFirst[k.var<const uint_>("group_id - 1")] << ";\n"
298  << " if(key == carry_in_key){\n"
299  << AssignVal{ValVar("other_value"), ValArrItem(carryInValuesFirst, "group_id - 1", vBytes, k), vBytes} << ";\n"
300  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
301  << " }\n"
302  << "}\n"
303  << "if (save) {\n"
304  << AssignKey{KeyVec(keysResult, "key", k), KeyVec(keys, "gid", k), nKeys} << ";\n"
305  << AssignVal{ValArrItem(valuesResult, "key", vBytes, k), ValVar("result"), vBytes} << ";\n"
306  << "}\n";
307 
308  auto workGroupsNo = static_cast<std::size_t>(
309  std::ceil(static_cast<float>(count) / static_cast<float>(workGroupSize)));
310 
311  const compute::context &context = queue.get_context();
312  compute::kernel kernel = k.compile(context);
313  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
314  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
315 
316  queue.enqueue_1d_range_kernel(kernel,
317  0,
318  workGroupsNo * workGroupSize,
319  workGroupSize);

339  inline std::size_t ReduceByKeyWithScan(const std::vector<std::reference_wrapper<const compute::vector<uint_>>> &keys,
340  const compute::vector<unsigned char> &values,
341  const std::vector<std::reference_wrapper<compute::vector<uint_>>> &keysResult,
342  compute::vector<unsigned char> &valuesResult,
343  std::size_t valueByteSize,
344  const std::string &reduceBody,
345  compute::command_queue &queue) {
346  const compute::context &context = queue.get_context();
347  const std::size_t count = keys.at(0).get().size();
348 
349  if (count == 0) {
350  return 0;
351  }
352 
353  const compute::device &device = queue.get_device();
354  const std::size_t workGroupSize = device.get_info<CL_DEVICE_MAX_WORK_GROUP_SIZE>();
355  compute::vector<uint_> newKeys(count, context);
356  GenerateUintKeys(keys, newKeys.begin(), workGroupSize, queue);
357 
358  const auto carryOutSize = static_cast<std::size_t>(
359  std::ceil(static_cast<float>(count) / static_cast<float>(workGroupSize)));
360  compute::vector<uint_> carryOutKeys(carryOutSize, context);
361  compute::vector<unsigned char> carryOutValues(carryOutSize * valueByteSize, context);
362  CarryOuts(newKeys,
363  values,
364  carryOutKeys.begin(),
365  carryOutValues.begin(),
366  workGroupSize,
367  valueByteSize,
368  reduceBody,
369  queue);
370 
371  compute::vector<unsigned char> carryInValues(carryOutSize * valueByteSize, context);
372  CarryIns(carryOutKeys.begin(),
373  carryOutValues.begin(),
374  carryInValues.begin(),
375  carryOutSize,
376  workGroupSize,
377  valueByteSize,
378  reduceBody,
379  queue);
380 
381  const std::size_t resultSize = compute::detail::read_single_value<uint_>(newKeys.get_buffer(), count - 1, queue) + 1;
382 
383  for (compute::vector<uint_> &kRes : keysResult) {
384  kRes.resize(resultSize, queue);
385  }
386  valuesResult.resize(resultSize * valueByteSize, queue);
387 
388  FinalReduction(keys,
389  values.begin(),
390  keysResult,
391  valuesResult.begin(),
392  count,
393  newKeys.begin(),
394  carryOutKeys.begin(),
395  carryInValues.begin(),
396  workGroupSize,
397  valueByteSize,
398  reduceBody,
399  queue);
400 
401  return resultSize;
402  }

512  inline std::size_t ReducePairKey(const boost::compute::vector<unsigned int> &inputIndices1,
513  const boost::compute::vector<unsigned int> &inputIndices2,
514  boost::compute::vector<unsigned int> &outputIndices1,
515  boost::compute::vector<unsigned int> &outputIndices2,
516  boost::compute::command_queue &queue) {
517  using namespace boost;
518 
519  compute::context ctx = queue.get_context();
520  assert(inputIndices1.size() == inputIndices2.size());
521  const std::size_t inputSize = inputIndices1.size();
522 
523  if (inputSize <= 1) {
524  outputIndices1.resize(inputSize, queue);
525  outputIndices2.resize(inputSize, queue);
526  compute::copy(inputIndices1.begin(), inputIndices1.end(), outputIndices1.begin(), queue);
527  compute::copy(inputIndices2.begin(), inputIndices2.end(), outputIndices2.begin(), queue);
528  return inputSize;
529  }
530 
531  compute::vector<unsigned int> outputPos(inputSize, ctx);
532  detail::GenerateUintKeys({std::ref(inputIndices1), std::ref(inputIndices2)},
533  outputPos.begin(),
534  queue.get_device().get_info<CL_DEVICE_MAX_WORK_GROUP_SIZE>(),
535  queue);
536 
537  const std::size_t uniqueNum = (outputPos.end() - 1).read(queue) + 1;
538  outputIndices1.resize(uniqueNum, queue);
539  outputIndices2.resize(uniqueNum, queue);
540 
541  BOOST_COMPUTE_CLOSURE(void, copyToOutput, (unsigned int i), (inputIndices1, inputIndices2, outputIndices1, outputIndices2, outputPos),
542  {
543  if (outputPos[i] != outputPos[i - 1]) {
544  outputIndices1[outputPos[i]] = inputIndices1[i];
545  outputIndices2[outputPos[i]] = inputIndices2[i];
546  }
547  });
548 
549  compute::for_each_n(compute::counting_iterator(1), inputSize - 1, copyToOutput, queue);
550  outputIndices1.begin().write(inputIndices1.begin().read(queue), queue);
551  outputIndices2.begin().write(inputIndices2.begin().read(queue), queue);
552 
553  return uniqueNum;
554  }
