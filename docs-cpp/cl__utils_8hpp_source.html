48  template<typename T>
49  bool sort_by_key(const cl::Buffer& uint_keys, const cl::Buffer& t_values, uint size, cl::CommandQueue& queue) {
50  auto& state = acquire_state<T>();
51  auto& sort = state.sort_bitonic;
52 
53  if (!sort.ensure(state.type)) return false;
54 
55  auto* acc = get_acc_cl();
56 
57  cl::NDRange global(acc->get_max_wgs());
58  cl::NDRange local(acc->get_max_wgs());
59 
60  if (size <= sort.local_size) {
61  sort.local.setArg(0, uint_keys);
62  sort.local.setArg(1, t_values);
63  sort.local.setArg(2, size);
64  queue.enqueueNDRangeKernel(sort.local, cl::NDRange(), global, local);
65  } else {
66  sort.global.setArg(0, uint_keys);
67  sort.global.setArg(1, t_values);
68  sort.global.setArg(2, size);
69  queue.enqueueNDRangeKernel(sort.global, cl::NDRange(), global, local);
70  }
71 
72  return true;
73  }
