112  template<typename T, int capacity>
113  void StorageManager<T, capacity>::validate_rw(Format format, Storage& storage) {
114  if (storage.is_valid(format)) {
115  return;
116  }
117  if (!storage.is_valid_any()) {
118  validate_wd(format, storage);
119  return;
120  }
121 
122  const int source = -1;
123  const int infinity = -2;
124  const int target = static_cast<int>(format);
125 
126  std::array<int, capacity> reached;
127  std::queue<int> queue;
128  reached.fill(infinity);
129 
130  for (int i = 0; i < capacity; ++i) {
131  if (storage.is_valid_i(i)) {
132  reached[i] = source;
133  queue.push(i);
134  }
135  }
136 
137  while (reached[target] == infinity) {
138  int u = queue.front();
139  queue.pop();
140 
141  for (const auto& target_format : m_convert_rules[u]) {
142  if (reached[target_format.first] == infinity) {
143  reached[target_format.first] = u;
144  queue.push(target_format.first);
145  }
146  }
147  }
148 
149  std::vector<std::pair<int, int>> path;
150  int current = target;
151 
152  while (reached[current] != source) {
153  path.push_back({reached[current], current});
154  current = reached[current];
155  }
156 
157  for (auto iter = path.rbegin(); iter != path.rend(); ++iter) {
158  std::pair<int, int> from_to = *iter;
159  auto search_predicate = [from_to](const std::pair<int, int>& rule) { return rule.first == from_to.second; };
160  auto rule = std::find_if(m_convert_rules[from_to.first].begin(), m_convert_rules[from_to.first].end(), search_predicate);
161 
162  if (storage.get_ref_i(from_to.second).is_null()) {
163  m_constructors[from_to.second](storage);
164  }
165 
166  m_converters[rule->second](storage);
167  storage.validate(static_cast<Format>(from_to.second));
168  }
169  }
