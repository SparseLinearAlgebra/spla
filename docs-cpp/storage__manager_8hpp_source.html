127  template<typename T, typename F, int capacity>
128  void StorageManager<T, F, capacity>::validate_rw(F format, Storage& storage) {
129  if (storage.is_valid(format)) {
130  return;
131  }
132  if (!storage.is_valid_any()) {
133  const int i = static_cast<int>(format);
134  if (!storage.get_ptr_i(i)) {
135  m_constructors[i](storage);
136  }
137  if (m_validators[i]) {
138  m_validators[i](storage);
139  }
140  storage.validate(format);
141  return;
142  }
143 
144  const int source = -1;
145  const int infinity = -2;
146  const int target = static_cast<int>(format);
147 
148  std::array<int, capacity> reached;
149  std::queue<int> queue;
150  reached.fill(infinity);
151 
152  for (int i = 0; i < capacity; ++i) {
153  if (storage.is_valid_i(i)) {
154  reached[i] = source;
155  queue.push(i);
156  }
157  }
158 
159  while (reached[target] == infinity) {
160  int u = queue.front();
161  queue.pop();
162 
163  for (const auto& target_format : m_convert_rules[u]) {
164  if (reached[target_format.first] == infinity) {
165  reached[target_format.first] = u;
166  queue.push(target_format.first);
167  }
168  }
169  }
170 
171  std::vector<std::pair<int, int>> path;
172  int current = target;
173 
174  while (reached[current] != source) {
175  path.push_back({reached[current], current});
176  current = reached[current];
177  }
178 
179  for (auto iter = path.rbegin(); iter != path.rend(); ++iter) {
180  std::pair<int, int> from_to = *iter;
181  auto search_predicate = [from_to](const std::pair<int, int>& rule) { return rule.first == from_to.second; };
182  auto rule = std::find_if(m_convert_rules[from_to.first].begin(), m_convert_rules[from_to.first].end(), search_predicate);
183 
184  if (storage.get_ref_i(from_to.second).is_null()) {
185  m_constructors[from_to.second](storage);
186  }
187 
188  m_converters[rule->second](storage);
189  storage.validate(static_cast<F>(from_to.second));
190  }
191  }
