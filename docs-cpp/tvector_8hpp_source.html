283  template<typename T>
284  void TVector<T>::decorator_ensure(Format format) {
285  switch (format) {
286  case Format::CpuDenseVec: {
287  auto p_vec = get_dec_or_create_p<CpuDenseVec<T>>();
288  if (p_vec->get_version() >= m_version) return;
289 
290  if (is_valid()) {
291 #if defined(SPLA_BUILD_OPENCL)
292  auto p_cl_acc = get_acc_cl();
293  auto p_cl_dense_vec = get_dec_p<CLDenseVec<T>>();
294 
295  assert(p_cl_dense_vec && p_cl_dense_vec->is_valid_version(m_version));
296 
297  cpu_dense_vec_resize(m_n_rows, *p_vec);
298  cl_dense_vec_read(m_n_rows, p_vec->Ax.data(), *p_cl_dense_vec, p_cl_acc->get_queue_default());
299 #endif
300  } else {
301  validate();
302  cpu_dense_vec_resize(m_n_rows, *p_vec);
303  }
304 
305  p_vec->update_version(m_version);
306  if (m_reduce) p_vec->reduce = m_reduce->function;
307  } break;
308  case Format::CpuCooVec: {
309  LOG_MSG(Status::NotImplemented, "not supported");
310  } break;
311 #if defined(SPLA_BUILD_OPENCL)
312  case Format::CLDenseVec: {
313  auto p_cpu_vec = get_dec_or_create_p<CpuDenseVec<T>>();
314  auto p_cl_vec = get_dec_or_create_p<CLDenseVec<T>>();
315  if (p_cl_vec->get_version() >= m_version) return;
316 
317  if (is_valid()) {
318  decorator_ensure(Format::CpuDenseVec);
319  cl_dense_vec_init(m_n_rows, p_cpu_vec->Ax.data(), *p_cl_vec);
320  } else {
321  validate();
322  cl_dense_vec_init<T>(m_n_rows, nullptr, *p_cl_vec);
323  }
324 
325  p_cl_vec->update_version(m_version);
326  } break;
327  case Format::CLCooVec: {
328  LOG_MSG(Status::NotImplemented, "not supported");
329  } break;
330 #endif
331  default:
332  break;
333  }
334  }
