41  void cpu_dok_vec_to_coo(const CpuDokVec<T>& in,
42  CpuCooVec<T>& out) {
43  assert(out.Ai.size() == in.values);
44  assert(out.Ax.size() == in.values);
45 
46  uint k = 0;
47 
48  for (const auto& entry : in.Ax) {
49  const uint i = entry.first;
50  const T x = entry.second;
51  out.Ai[k] = i;
52  out.Ax[k] = x;
53  k += 1;
54  }
55 
56  out.values = in.values;
57  }
