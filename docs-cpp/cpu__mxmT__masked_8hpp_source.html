99

101  const auto& B_lst = p_lil_B->Ar[mask_i];
102 
103  auto A_it = A_lst.begin();

106  const auto B_end = B_lst.end();
107

109  if (A_it->first == B_it->first) {

111  ++A_it;

113  } else if (A_it->first < B_it->first) {
114  ++A_it;
115  } else {
116  ++B_it;
117  }
118  }
119  }

122  }
