73  auto* p_cl_v_dense = v->template get<CLDenseVec<T>>();

79 
80  m_kernel.setArg(0, p_cl_v_dense->Ax);
81  m_kernel.setArg(1, cl_count);

85 
86  cl::NDRange global(m_block_size * n_groups_to_dispatch);
87  cl::NDRange local(m_block_size);
88  queue.enqueueNDRangeKernel(m_kernel, cl::NDRange(), global, local);
89 
90  queue.enqueueReadBuffer(cl_count, true, 0, sizeof(count[0]), count);
91  r->set_int(count[0]);

93  return Status::Ok;

102  CLProgramBuilder program_builder;

108 
109  if (!program_builder.build()) return false;
