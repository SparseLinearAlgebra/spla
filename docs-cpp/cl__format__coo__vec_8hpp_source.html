87  void cl_coo_vec_read(const std::size_t n_values,
88  uint* Ai,
89  T* Ax,
90  const CLCooVec<T>& storage,
91  cl::CommandQueue& queue,
92  bool blocking = true) {
93  if (n_values == 0) {
94  LOG_MSG(Status::Ok, "nothing to do");
95  return;
96  }
97 
98  const std::size_t buffer_size_Ai = n_values * sizeof(uint);
99  const std::size_t buffer_size_Ax = n_values * sizeof(T);
100  const auto flags = CL_MEM_READ_ONLY | CL_MEM_HOST_READ_ONLY | CL_MEM_ALLOC_HOST_PTR;
101 
102  cl::Buffer staging_Ai(get_acc_cl()->get_context(), flags, buffer_size_Ai);
103  cl::Buffer staging_Ax(get_acc_cl()->get_context(), flags, buffer_size_Ax);
104 
105  queue.enqueueCopyBuffer(storage.Ai, staging_Ai, 0, 0, buffer_size_Ai);
106  queue.enqueueCopyBuffer(storage.Ax, staging_Ax, 0, 0, buffer_size_Ax);
107  queue.enqueueReadBuffer(staging_Ai, blocking, 0, buffer_size_Ai, Ai);
108  queue.enqueueReadBuffer(staging_Ax, blocking, 0, buffer_size_Ax, Ax);
109  }

111  template<typename T>
112  void cl_coo_vec_to_dense(const std::size_t n_rows,
113  const std::size_t n_values,
114  const CLCooVec<T>& in,
115  CLDenseVec<T>& out,
116  cl::CommandQueue& queue) {
117  CLProgramBuilder builder;
118  builder.set_name("vector_format")
119  .add_type("TYPE", get_ttype<T>().template as<Type>())
120  .set_source(source_vector_formats);
121 
122  if (!builder.build()) return;
123 
124  cl_fill_zero<T>(queue, out.Ax, n_rows);
125 
126  auto* acc = get_acc_cl();
127 
128  uint block_size = acc->get_wave_size();
129  uint n_groups_to_dispatch = std::max(std::min(in.values / block_size, uint(512)), uint(1));
130 
131  auto kernel = builder.make_kernel("sparse_to_dense");
132  kernel.setArg(0, in.Ai);
133  kernel.setArg(1, in.Ax);
134  kernel.setArg(2, out.Ax);
135  kernel.setArg(3, uint(in.values));
136 
137  cl::NDRange global(block_size * n_groups_to_dispatch);
138  cl::NDRange local(block_size);
139  queue.enqueueNDRangeKernel(kernel, cl::NDRange(), global, local);
140  queue.finish();
141  }
