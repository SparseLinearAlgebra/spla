57  void cpu_lil_add_element(uint row_id,
58  uint col_id,
59  T element,
60  CpuLil<T>& lil) {
61  using Entry = typename CpuLil<T>::Entry;
62  auto& row = lil.Ar[row_id];
63 
64  auto where = std::upper_bound(row.begin(), row.end(), Entry{col_id, element}, [](const Entry& val, const Entry& point) {
65  return val.first < point.first;
66  });
67 
68  if (where != row.end() && (*where).first == col_id) {
69  (*where).second = lil.reduce((*where).second, element);
70  return;
71  }
72 
73  row.insert(where, Entry{col_id, element});
74  lil.values += 1;
75  }
