96  void CpuLil<T>::add_element(uint row_id, uint col_id, T element) {
97  assert(row_id < n_rows);
98  assert(col_id < n_cols);
99 
100  Row &row = data[row_id];
101  uint i = 0;
102 
103  auto where = std::upper_bound(row.begin(), row.end(), Entry{col_id, element}, [](const Entry &val, const Entry &point) {
104  return val.first < point.first;
105  });
106 
107  if (where != row.end() && *where.first == col_id) {
108  *where.second = reduce(*where.second, element);
109  return;
110  }
111 
112  row.insert(where, Entry{col_id, element});
113  n_values += 1;
114  }
