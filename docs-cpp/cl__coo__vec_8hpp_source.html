68

70  cl::Buffer staging_Ax(get_acc_cl()->get_context(), flags, buffer_size_Ax);
71 
72  queue.enqueueCopyBuffer(storage.Ai, staging_Ai, 0, 0, buffer_size_Ai);
73  queue.enqueueCopyBuffer(storage.Ax, staging_Ax, 0, 0, buffer_size_Ax);
74  queue.enqueueReadBuffer(staging_Ai, blocking, 0, buffer_size_Ai, Ai);
75  queue.enqueueReadBuffer(staging_Ax, blocking, 0, buffer_size_Ax, Ax);
