139  ref_ptr<TDecoration<T>>& TMatrix<T>::get_dec_or_create(Format format) {
140  auto index = static_cast<int>(format);
141 
142  if (m_decorations[index]) {
143  return m_decorations[index];
144  }
145 
146  if (format == Format::CpuLil) {
147  return m_decorations[index] = make_ref<CpuLil<T>>();
148  }
149  if (format == Format::CpuDok) {
150  return m_decorations[index] = make_ref<CpuDok<T>>();
151  }
152  if (format == Format::CpuCoo) {
153  return m_decorations[index] = make_ref<CpuCoo<T>>();
154  }
155  if (format == Format::CpuCsr) {
156  return m_decorations[index] = make_ref<CpuCsr<T>>();
157  }
158  if (format == Format::CpuCsc) {
159  return m_decorations[index] = make_ref<CpuCsr<T>>();
160  }
161 
162  LOG_MSG(Status::NotImplemented, "unable to create decoration of specified format");
163  return m_decorations.back();

259  template<typename T>
260  void TMatrix<T>::ensure_lil_format() {
261  auto p_lil = get_dec_or_create_p<CpuLil<T>>();
262 
263  if (p_lil->get_version() < m_version) {
264  LOG_MSG(Status::Error, "data invalidation, previous content lost");
265  update_version();
266  cpu_lil_resize(m_n_rows, *p_lil);
267  p_lil->update_version(m_version);
268  }
269  }

271  template<typename T>
272  void TMatrix<T>::ensure_dok_format() {
273  auto p_dok = get_dec_or_create_p<CpuDok<T>>();
274 
275  if (p_dok->get_version() < m_version) {
276  auto p_lil = get_dec_p<CpuLil<T>>();
277 
278  assert(!p_lil || p_lil->get_version() == m_version && "only lil read supported");
279 
280  if (p_lil && p_lil->get_version() == m_version) {
281  LOG_MSG(Status::Ok, "copy data from lil, preserve content");
282  cpu_lil_to_dok(m_n_rows, *p_lil, *p_dok);
283  }
284 
285  p_dok->update_version(m_version);
286  }
