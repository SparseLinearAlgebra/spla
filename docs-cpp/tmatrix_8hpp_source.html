307  template<typename T>
308  void TMatrix<T>::ensure_lil_format() {
309  auto p_lil = get_dec_or_create_p<CpuLil<T>>();
310 
311  if (p_lil->get_version() < m_version) {
312  if (is_valid()) {
313  // todo: sync data
314  LOG_MSG(Status::Error, "data invalidation, previous content lost");
315  update_version();// todo: keep
316  cpu_lil_resize(m_n_rows, *p_lil);
317  } else {
318  validate();
319  update_version();
320  cpu_lil_resize(m_n_rows, *p_lil);
321  }
322 
323  p_lil->update_version(m_version);
324  if (m_reduce) p_lil->reduce = m_reduce->function;
325  }
326  }

328  template<typename T>
329  void TMatrix<T>::ensure_dok_format() {
330  auto p_dok = get_dec_or_create_p<CpuDok<T>>();
331 
332  if (p_dok->get_version() < m_version) {
333 
334  if (is_valid()) {
335  auto p_lil = get_dec_p<CpuLil<T>>();
336  assert(!p_lil || p_lil->get_version() == m_version && "only lil read supported");
337 
338  if (p_lil && p_lil->get_version() == m_version) {
339  LOG_MSG(Status::Ok, "copy data from lil, preserve content");
340  cpu_lil_to_dok(m_n_rows, *p_lil, *p_dok);
341  }
342  }
343 
344  p_dok->update_version(m_version);
345  if (m_reduce) p_dok->reduce = m_reduce->function;
346  }
