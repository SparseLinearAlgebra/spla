294  BOOST_STATIC_ASSERT(compute::is_device_iterator<OutputIterator2>::value);

297  typedef typename std::iterator_traits<InputIterator3>::value_type value_type;
298 
299  int tile_size = 1024;
300 
301  int count1 = compute::detail::iterator_range_size(maskFirst, maskLast);
302  int count2 = compute::detail::iterator_range_size(keyFirsts, keyLast);
303 
304  compute::vector<compute::uint_> tile_a((count1 + count2 + tile_size - 1) / tile_size + 1, queue.get_context());
305  compute::vector<compute::uint_> tile_b((count1 + count2 + tile_size - 1) / tile_size + 1, queue.get_context());
306 
307  // Tile the sets
308  BalancedPathKernel tiling_kernel;
309  tiling_kernel.tile_size = tile_size;
310  tiling_kernel.set_range(maskFirst, maskLast, keyFirsts, keyLast,
311  tile_a.begin() + 1, tile_b.begin() + 1, compare);
312  fill_n(tile_a.begin(), 1, 0, queue);
313  fill_n(tile_b.begin(), 1, 0, queue);
314  tiling_kernel.exec(queue);
315 
316  fill_n(tile_a.end() - 1, 1, count1, queue);
317  fill_n(tile_b.end() - 1, 1, count2, queue);
318 
319  compute::vector<key_type> temp_resultKeys(count1 + count2, queue.get_context());
320  compute::vector<value_type> temp_resultValues(count1 + count2, queue.get_context());
321  compute::vector<compute::uint_> counts((count1 + count2 + tile_size - 1) / tile_size + 1, queue.get_context());
322  compute::fill_n(counts.end() - 1, 1, 0, queue);
323 
324  // Find individual intersections
325  SerialIntersectionKernel intersection_kernel;
326  intersection_kernel.tile_size = tile_size;
327  intersection_kernel.set_range(maskFirst, keyFirsts, tile_a.begin(), tile_a.end(),
328  tile_b.begin(), valueFirst,
329  temp_resultKeys.begin(),
330  temp_resultValues.begin(),
331  counts.begin(),
332  compare, equals);
333 
334  intersection_kernel.exec(queue);
335 
336  exclusive_scan(counts.begin(), counts.end(), counts.begin(), queue);
337 
338  // Compact the results keys
339  compute::detail::compact_kernel compactKey_kernel;
340  compactKey_kernel.tile_size = tile_size;
341  compactKey_kernel.set_range(temp_resultKeys.begin(), counts.begin(), counts.end(), resultKeys);
342  compactKey_kernel.exec(queue);
343 
344  // Compact the results values
345  compute::detail::compact_kernel compactValues_kernel;
346  compactValues_kernel.tile_size = tile_size;
347  compactValues_kernel.set_range(temp_resultValues.begin(), counts.begin(), counts.end(), resultValues);
348  compactValues_kernel.exec(queue);
349 
350  return (counts.end() - 1).read(queue);
