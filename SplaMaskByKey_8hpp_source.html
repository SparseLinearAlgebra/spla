309  template<
310  typename InputMask,
311  typename InputKey,
312  typename Compare,
313  typename Equals,
314  typename ResizeResult,
315  typename AssignResult>
316  std::size_t MaskByKey(InputMask maskFirst,
317  InputKey keyFirst,
318  Compare compare,
319  Equals equals,
320  ResizeResult resizeResult,
321  AssignResult assignResult,
322  std::size_t maskCount,
323  std::size_t keyCount,
324  bool complement,
325  boost::compute::command_queue &queue) {
326  using namespace boost;
327  typedef typename std::iterator_traits<InputMask>::value_type key_type;
328 
329  std::size_t tileSize = 1024;
330 
331  compute::vector<compute::uint_> tileA((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
332  compute::vector<compute::uint_> tileB((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
333 
334  // Tile the sets
335  detail::BalancedPathKernel balancedPathKernel;
336  balancedPathKernel.tile_size = tileSize;
337  balancedPathKernel.set_range(maskFirst, maskFirst + maskCount,
338  keyFirst, keyFirst + keyCount,
339  tileA.begin() + 1, tileB.begin() + 1,
340  compare);
341  fill_n(tileA.begin(), 1, 0, queue);
342  fill_n(tileB.begin(), 1, 0, queue);
343  balancedPathKernel.exec(queue);
344 
345  fill_n(tileA.end() - 1, 1, maskCount, queue);
346  fill_n(tileB.end() - 1, 1, keyCount, queue);
347 
348  compute::vector<compute::uint_> counts((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
349  compute::fill_n(counts.end() - 1, 1, 0, queue);
350 
351  // Find result intersections count and offsets to write result of each tile
352  detail::SerialIntersectionCountKernel intersectionCountKernel;
353  intersectionCountKernel.set_range(maskFirst, keyFirst,
354  tileA.begin(), tileA.end(),
355  tileB.begin(),
356  counts.begin(),
357  compare, equals, complement);
358  intersectionCountKernel.exec(queue);
359 
360  // Compute actual counts offsets
361  exclusive_scan(counts.begin(), counts.end(), counts.begin(), queue);
362 
363  // Get result count and resize buffers
364  std::size_t resultCount = (counts.end() - 1).read(queue);
365  resizeResult(resultCount);
366 
367  // Find result intersections
368  detail::SerialIntersectionKernel intersectionKernel;
369  intersectionKernel.set_range(maskFirst, keyFirst, tileA.begin(), tileA.end(),
370  tileB.begin(),
371  counts.begin(),
372  compare, equals, assignResult, complement);
373  intersectionKernel.exec(queue);
374 
375  return resultCount;
376  }
