51  inline void SortByRow(boost::compute::vector<unsigned int> &rows,
52  boost::compute::vector<unsigned char> &vals,
53  std::size_t elementsInSequence,
54  boost::compute::command_queue &queue) {
55  using namespace boost;
56 
57  compute::context ctx = queue.get_context();
58  std::size_t nvals = rows.size();
59  bool typeHasValues = elementsInSequence != 0;
60 
61  assert(vals.size() == nvals * elementsInSequence);
62 
63  // Use permutation buffer to synchronize position of indices of the same
64  // entries in all 2 buffers: rows and vals.
65  compute::vector<unsigned int> permutation(nvals, ctx);
66  compute::copy(compute::counting_iterator<unsigned int>(0),
67  compute::counting_iterator<unsigned int>(nvals),
68  permutation.begin(),
69  queue);
70 
71  // Sort in row order and then, using permutation, shuffle column indices
72  compute::sort_by_key(rows.begin(), rows.end(), permutation.begin(), queue);
73 
74  // Copy values, using permutation
75  if (typeHasValues) {
76  compute::vector<unsigned char> valsTmp(nvals * elementsInSequence, ctx);
77  Gather(permutation.begin(), permutation.end(), vals.begin(), valsTmp.begin(), elementsInSequence, queue);
78  std::swap(vals, valsTmp);
79  }
80  }
